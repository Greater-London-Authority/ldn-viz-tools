# Treemaps

Treemaps are used for hierarchical data that have a part-to-whole relationship.

We prefer treemaps with centred text as they work best across different screen sizes. These are ideal for more infographic-type treemaps.

However, for treemaps with more text density, it may be preferable to use treemaps with text aligned to the top left. There are two examples, one with one text mark which is simple to reproduce. The second has variable text positioning dependent on other visible text, to ensure the first line of text is always in the same position.

### Centred Text

<details>
	<summary class="cursor-pointer dark:text-color-text-inverse-primary">Show data processing</summary>
	```javascript
    let width = $state(0);
	const height = $derived(width * (6 / 16));

    const toTitleCase = (str: string) => {
    	return str
    		.toLowerCase()
    		.split(' ')
    		.map((word: string) => word.charAt(0).toUpperCase() + word.slice(1))
    		.join(' ');
    };

    const group = (data: any, [name, ...path]: string[]): any => {
    	return d3
    		.groups(data, (d) => d[name])
    		.map(([value, rows]) =>
    			path.length
    				? { name: value, children: group(rows, path) }
    				: { name: value, value: rows.length }
    		);
    };

    const root = { name: 'root', children: group(penguins, ['island', 'species', 'sex']) };

    const leaves = (root: any, width: number, height: number) => {
    	const hierarchy = d3
    		.hierarchy(root)
    		.sum((d) => d.value)
    		.sort((a, b) => (b.value ?? 0) - (a.value ?? 0));
    	d3.treemap().size([width, height]).padding(1)(hierarchy);
    	return hierarchy.leaves();
    };

    const leafData = $derived.by(() => leaves(root, width, height));
    ```

### Top Left Text - Variable Text Position

<details>
	<summary class="cursor-pointer dark:text-color-text-inverse-primary">Show data processing</summary>
	```javascript
    let width = $state(0);
	const height = $derived(width * (6 / 16));

    // sanitising data for display by changing sex from uppercase to title case
    const toTitleCase = (str: string) => {
    	return str
    		.toLowerCase()
    		.split(' ')
    		.map((word: string) => word.charAt(0).toUpperCase() + word.slice(1))
    		.join(' ');
    };

    const penguinsPreprocessed = penguins.map((d) => ({
    	...d,
    	sex: d.sex ? toTitleCase(d.sex) : 'N/A'
    }));

    // turning tidy data into hierarchical data for use in treemap
    const group = (data: any, [name, ...path]: string[]): any => {
    	return d3
    		.groups(data, (d) => d[name])
    		.map(([value, rows]) =>
    			path.length
    				? { name: value, children: group(rows, path) }
    				: { name: value, value: rows.length }
    		);
    };

    const root = {
    	name: 'root',
    	children: group(penguinsPreprocessed, ['island', 'species', 'sex'])
    };

    const leaves = (root: any, width: number, height: number) => {
    	const hierarchy = d3
    		.hierarchy(root)
    		.sum((d) => d.value)
    		.sort((a, b) => (b.value ?? 0) - (a.value ?? 0));
    	d3.treemap().size([width, height]).padding(1)(hierarchy);
    	return hierarchy.leaves();
    };

    const leafData = $derived.by(() => leaves(root, width, height));

    // function checking whether text should be displayed and which text should be displayed
    const showText = (text: string, d: any, n: number) => {
    	const w = text.toString().length * 12 + 5;
    	const h = 20 + 20 * n;
    	const visible = d.x1 - d.x0 > w && d.y1 - d.y0 > h;
    	return visible;
    };

    const preprocessTextVisible = (data: any) => {
    	return data.map((d: any) => ({
    		...d,
    		visible: {
    			species: showText(d.parent.data.name, d, 2),
    			sex: showText(d.data.name, d, 1),
    			value: showText(d.value, d, 1)
    		}
    	}));
    };

    const preprocessedData = $derived(preprocessTextVisible(leafData));
    ```
